# noinspection pyUnusedParameter
def reconstruct_inc(
        incid_past: Union[np.ndarray, pd.Series],
        rt_fore_array: Union[np.ndarray, pd.Series, pd.DataFrame],
        tg_array, tg_max=None,
        incid_fore=None, seed=None, **kwargs):
    """Constructs incidence time series with the renewal equation.

    Requires granular data on past incidence, past and future
    generation time, and future reproduction number. The shape of the
    output incidence series (`ct_fore`) is based on the shape
    of the reproduction number array (`rt_fore_array`).

    Parameters
    ----------
    incid_past : np.ndarray, pd.Series
        Incidence time series in the past.
    rt_fore_array : np.ndarray, pd.Series, pd.DataFrame
        A time series with \\(R(t)\\) in future periods. It can be a
        1D or 2D array like, with the following signatures:

        - 1D: `a[i_t]`
        - 2D: `a[i_sample, i_t]`
    tg_array : ???
        Generation time distribution.
        TODO expand compatibility for time-dependent array
    incid_fore : optional
        Output container. Holds the future incidence series.
        The shape and signature is the same as `rt_fore_array`.
    tg_max :
        Truncation value (one past) for generation time.
    seed : int, optional

    Returns
    -------
    incid_fore : np.ndarray
        An array with the incidence series. The signature and shape are
        the same as the input \\(R(t)\\) array, `rt_fore_array`.
    """
    num_steps_fore = rt_fore_array.shape[-1]

    # Input types interpretation
    # --------------------------

    # --- R(t) input
    run_mult = False  # Whether to run the reconstruction over multiple R(t) series.
    if isinstance(rt_fore_array, np.ndarray):  # Numpy array
        if rt_fore_array.ndim == 2:
            run_mult = True
        elif rt_fore_array.ndim > 2:
            raise ValueError(f"Hey, input R(t) array rt_fore must either be 1D or 2D (ndim = {rt_fore_array.ndim}).")

    elif isinstance(rt_fore_array, pd.Series):  # Pandas 1D series
        rt_fore_array = rt_fore_array.to_numpy()

    elif isinstance(rt_fore_array, pd.DataFrame):  # Pandas 2D data frame
        rt_fore_array = rt_fore_array.to_numpy()
        run_mult = True

    else:
        raise TypeError(f"Hey, invalid type for input R(t) rt_fore: {type(rt_fore_array)}")

    # --- C(t) past array
    if isinstance(incid_past, pd.Series):
        incid_past = incid_past.to_numpy()

    # --- Tg(s) array
    if isinstance(tg_array, pd.Series):
        tg_array = tg_array.to_numpy()

    # Optional arguments handling
    # ---------------------------

    if tg_max is None:
        tg_max = tg_array.shape[0]

    # Output array
    if incid_fore is None:
        incid_fore = np.empty_like(rt_fore_array, dtype=int)
        #     ^  ^ Agnostic to 1D or 2D

    # Time-based seed
    if seed is None:
        seed = round(1000 * time.time())

    # Dispatch and run the reconstruction
    # -----------------------------------
    if run_mult:
        return reconstruct_ct_multiple(
            incid_past, rt_fore_array, tg_array, tg_max, incid_fore,
            num_steps_fore, seed)
    else:
        return reconstruct_ct_single(
            incid_past, rt_fore_array, tg_array, tg_array, tg_max,
            incid_fore, num_steps_fore, seed)


@nb.njit
def reconstruct_ct_single(
        ct_past, rt_fore, tg_past, tg_fore, tg_max, ct_fore,
        num_steps_fore, seed):
    """DOCSTRINGS TODO

    NOTE: tg is assumed as constant, but different past and fore
    arrays are accepted.
    """

    np.random.seed(seed)  # Seeds the numba or numpy generator

    # Main loop over future steps
    for i_t_fut in range(num_steps_fore):  # Loop over future steps
        lamb = 0.  # Sum of generated cases from past cases
        # r_curr = rt_fore.iloc[i_t_fut]  # Pandas
        r_curr = rt_fore[i_t_fut]

        # Future series chunk
        for st in range(1, min(i_t_fut + 1, tg_max)):
            lamb += r_curr * tg_fore[st] * ct_fore[i_t_fut - st]

        # Past series chunk
        for st in range(i_t_fut + 1, tg_max):
            lamb += r_curr * tg_past[st] * ct_past[-(st - i_t_fut)]

        # Poisson number
        ct_fore[i_t_fut] = np.random.poisson(lamb)

    return ct_fore
