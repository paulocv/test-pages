from typing import Union

import numpy as np
import pandas as pd

from rtrend_forecast.utils import WEEK_DT, WEEK_REF_START


def get_period_unique_id(tlabel, dt=WEEK_DT, ref_start=WEEK_REF_START):
    """Returns an arbitrary ID for the period to which a given
    time label belongs.
    """
    return (tlabel - ref_start) // dt


def get_aggr_label_from_id(
        i_t, dt=WEEK_DT, ref_start=WEEK_REF_START):#, label_wkday=WEEKDAY_TGT):
    """Get the date timestamp that represents an aggr period,
    given its unique ID produced by `get_period_unique_id` with the
    same ref_start.
    """
    a = (i_t + 1) * dt + ref_start
    return a

def aggregate_in_periods(
        gran_sr: Union[pd.Series, pd.DataFrame], agg_dt: pd.Timedelta,
        ref_start: pd.Timestamp, full_only=True):
    """Combines granular time series into aggregated periods.
    Receives and returns Pandas objects (Series, DataFrame).

    1D signature (Series):    data_daily[tlabel]
    2D signature (DataFrame): data_daily.loc[i_sample, tlabel]

    Parameters
    ----------
    gran_sr : pd.Series, pd.DataFrame
        The granular data to be aggregated over time periods.
    agg_dt : pd.Timedelta
        The period length by which data must be aggregated.
    ref_start : pd.Timestamp
        Reference timestamp for the start of the aggregated period.
    full_only : bool, optional
        Whether to use return only complete aggregated periods.
        If `False`, periods for which the granular data is not complete
        are discarded. TODO NOT IMPLEMENTED YET.

    Returns
    -------
    TODO

    Notes
    -----
    DEVNOTE: A refurbishing of the 2D signature would make it much
    easier to set the time axis. I'll just have to use axis=0 instead
    of multiple methods to dynamically select an axis.
    Afterwards, it is enough to choose the other axis for quantile
    calculation and etc.
    """
    #
    axis = 1 if isinstance(gran_sr, pd.DataFrame) else 0
    if not isinstance(ref_start, pd.Timestamp):
        ref_start = pd.Timestamp(ref_start)

    # Group by agg periods
    gp = gran_sr.groupby(
        lambda t: get_period_unique_id(t, dt=agg_dt, ref_start=ref_start),
        axis=axis
    )

    # # Problem: not enough info for this step. Do it outside
    # if full_only:  # Filter by group size and regroup
    #     gp = gp.filter(lambda df: df.shape[0] == WEEKLEN).groupby(get_period_unique_id)

    res: Union[pd.Series, pd.DataFrame] = gp.sum()

    # # Make the fore tlabels index and set as the time axis
    new_idx = res.axes[axis].map(
        lambda i_t: get_aggr_label_from_id(
            i_t, dt=agg_dt, ref_start=ref_start))

    # Set the labels: compatibility with
    try:
        res.set_axis(new_idx, axis=axis, inplace=True)
    except TypeError:
        res.set_axis(new_idx, axis=axis, copy=False)  # pd >= 1.5.0

    return res
