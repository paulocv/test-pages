class ColoredTerminalHandler(logging.StreamHandler):
    """A custom logging handler to print color-coded messages to
    the terminal.
    """
    # --- Formatters
    # Default message: no color.
    _formatters = defaultdict(lambda: Formatter(
        term_basic_fmt, style="{"))

    # TODO: decide between Formatter or another custom one
    # Critical Error: red
    _formatters[logging.CRITICAL] = Formatter(
        red(term_basic_fmt), style="{")
    # _formatters[logging.CRITICAL] = RtrendTerminalFormatter(
    #     color=red)

    # Error: red
    _formatters[logging.ERROR] = Formatter(
        red(term_basic_fmt), style="{")
    # _formatters[logging.ERROR] = RtrendTerminalFormatter(
    #     color=red)

    # Warning: yellow
    _formatters[logging.WARNING] = Formatter(
        yellow(term_basic_fmt), style="{")

    # Success: green
    _formatters[SUCCESS] = Formatter(
        green(term_basic_fmt), style="{"
    )

    def format(self, record):
        """
        Format the specified record.
        Chooses color by log level, using the _formatters dict.
        """
        fmtr = self._formatters[record.levelno]
        # Create `last_name` field, which removes all parent names
        record.last_name = record.name.rsplit(".", 1)[-1]

        return fmtr.format(record)

class ExecTimeTracker:

    SEP = "."  # Separator of category names

    def __init__(self, category=None):

        # TODO: if category isnot None, preppend it to all measures
        self._dict = defaultdict(lambda: 0.)

    def get_dict(self):
        return self._dict

    def track(self, name=None, category=None):
        """Adds the decorated function to the execution time tracking
        collection.

        Parameters
        ----------
        name : str, optional
            Name to use as key in the dictionary of tracked functions.
            If not informed, uses the name of the function
        category : str, optional
            A string prepended to name. Works for nesting tracked
            functions.
        """

        def track_decorator(func):

            nonlocal name

            if not TRACK_EXEC_TIME:
                # Make dummy decorator
                @functools.wraps(func)
                def just_the_func(*args, **kwargs):
                    return func(*args, **kwargs)
                return just_the_func

            # Manipulate the tracking function name.
            if name is None:
                name = func.__name__
            if category is not None:
                name = self.SEP.join([category, name])

            # Define the time-measuring wrapper.
            @functools.wraps(func)
            def tracked_func(*args, **kwargs):
                # Run the function
                xt0 = time.time()
                res = func(*args, **kwargs)
                xtf = time.time()

                # Log the execution time
                self._dict[name] += xtf - xt0

                return res

            return tracked_func

        return track_decorator

    def join_categories(self):
        pass

    def merge_with(self, other):
        pass

    def report(self):
        """Returns a summary string with execution times, nested by
        categories.
        """
