import functools

import numpy as np
from pathos.pools import ProcessPool

WEEKLEN = 7  # Constant: number of days in a week
DAY_DT = pd.Timedelta("1d")
WEEK_DT = pd.Timedelta("1w")
WEEK_REF_START = pd.Timestamp(2022, 11, 13, 0)  # Arbitrary Sunday


def rsetattr(obj, attr, val):
    """Recursive setattr alternative that works with nested attributes.
    https://stackoverflow.com/questions/31174295/getattr-and-setattr-
    on-nested-subobjects-chained-properties/31174427#31174427
    """
    pre, _, post = attr.rpartition('.')
    return setattr(rgetattr(obj, pre) if pre else obj, post, val)


def rgetattr(obj, attr, *args):
    """Recursive getattr alternative that works with nested attributes.
    https://stackoverflow.com/questions/31174295/getattr-and-setattr-
    on-nested-subobjects-chained-properties/31174427#31174427
    """
    def _getattr(obj_, attr_):
        return getattr(obj_, attr_, *args)
    return functools.reduce(_getattr, [obj] + attr.split('.'))


def dateindex_to_floatrange(index: pd.DatetimeIndex):
    """Converts a pandas DatetimeIndex into an array of floats.
    The values are normalized to the [0, 1] range.
    """
    _min, _max = index.min(), index.max()
    assert _max - _min > pd.Timedelta(0)
    return (index - _min) / (_max - _min)


def dateindex_is_regular(index: pd.DatetimeIndex):
    """Return True if a pandas DatetimeIndex object is a
    regular grid of dates.
    """
    deltas = index.to_series().diff()
    return (deltas.iloc[2:] == deltas.iloc[1]).all()


def get_time_bins(
        idx: pd.DatetimeIndex, t0, dt, nperiods):
    """Get indexes of periodic bins from a time label index."""
    return ((idx - t0) // dt) % nperiods


def get_or_seed_rng(rng):
    """If an integer is passed, convert it to a default numpy random
    number generator
    """
    if isinstance(rng, int):
        rng = np.random.default_rng(rng)

    elif rng is None:  # Get unseeded generator
        rng = np.random.default_rng()

    return rng


def make_dir_from(pathname):
    """Creates directories recursively for a given path
    (possibly to a file).

    No message or error if some directories already exist.
    """
    dn = os.path.dirname(pathname)
    if dn:
        os.makedirs(dn, exist_ok=True)  # OS independent


def get_arg_or_kwarg(args, kwargs, i_arg, key):
    """Get a required argument that can also be passed as keyword
    argument (using *args and **kwargs).
    """

    try:
        val = args[i_arg]
    except IndexError:
        try:
            val = kwargs[key]
        except KeyError:
            raise TypeError(f"Hey, missing required positional"
                            f" argument: '{key}")

    return val


def extract_prefixed_keys(d: dict, prefix: str, remove_prefix=True):
    """Obtain entries from a dictionary whose keys have a given
    prefix.
    If `remove_prefix` is True, the resulting dictonary has the prefix
    removed from each key.
    """
    new_prefix = "" if remove_prefix else prefix
    return {key.replace(prefix, new_prefix): val
            for key, val in d.items()
            if prefix in key}


def rotate_ax_labels(ax, angle=60, xy="x", which="major"):
    """This function could be included in my plot_tools."""
    labels = (
        ax.get_xticklabels(which=which)
        if xy == "x" else ax.get_yticklabels(which=which))

    for label in labels:
        label.set(rotation=angle, horizontalalignment='right')


def make_tmp_unique_index():
    """Creates a unique integer index that can be appended to temporary
    paths.
    """
    maxint = np.iinfo(np.int).max
    try:
        make_tmp_unique_index.id += 1
    except AttributeError:
        rng = np.random.default_rng()  # Unseeded random generator
        make_tmp_unique_index.id = rng.integers(maxint)

    return make_tmp_unique_index.id


def map_parallel_or_sequential(task, contents, ncpus=1, pool=None):
    """
    Maps a function (task) into an iterable of inputs (contents), using
    either a sequential loop (if ncpus=1) or a pool of concurrent
    processes.
    Alternatively, for running in a pool of processes, a previously
    created pool can be informed, in which case ncpus is ignored.

    Parameters
    ----------
    task : callable
        A single-argument function to map on inputs.
    contents : iterable
        An iterable or sequence of inputs, each one will be passed
        as the only argument of 'task'.
    ncpus : int
        Number of concurrent processes. If ncpus=1 (default), a simple
         for loop is used. If greater than 1, a process
        pool is created. If parameter 'pool' is informed,
        ncpus is ignored.
    pool : Any
        A pathos process pool previously created. Overrides parameter
        'ncpus' if informed.
    """
    if ncpus == 1 and pool is None:
        # Run sequentially
        results = list()
        for item in contents:
            results.append(task(item))

        return results

    if pool is None:
        pool = ProcessPool(ncpus=ncpus)

    # Run parallel pool
    return pool.map(task, contents)


def os_open(fname):
    """Opens given file with the default OS application
    (extension-based).
    """
    try:
        # Unix
        subprocess.run(["open", fname], check=True)
    except subprocess.CalledProcessError:
        try:
            # Windows
            subprocess.run(["start", fname], check=True, shell=True)
        except subprocess.CalledProcessError:
            print("Unable to open the file with the default application.")
